92      void register(NioSocketChannel channel, ChannelFuture future) {
93  
94          boolean server = !(channel instanceof NioClientSocketChannel);
95          Runnable registerTask = new RegisterTask(channel, future, server);
96          Selector selector;
97  
98          synchronized (startStopLock) {
99              if (!started) {
100                 // Open a selector if this worker didn't start yet.
101                 try {
102                     this.selector = selector = Selector.open();
103                 } catch (Throwable t) {
104                     throw new ChannelException(
105                             "Failed to create a selector.", t);
106                 }
107 
108                 // Start the worker thread with the new Selector.
109                 String threadName =
110                     (server ? "New I/O server worker #"
111                             : "New I/O client worker #") + bossId + '-' + id;
112 
113                 boolean success = false;
114                 try {
115                     DeadLockProofWorker.start(
116                             executor, new ThreadRenamingRunnable(this, threadName));
117                     success = true;
118                 } finally {
119                     if (!success) {
120                         // Release the Selector if the execution fails.
121                         try {
122                             selector.close();
123                         } catch (Throwable t) {
124                             logger.warn("Failed to close a selector.", t);
125                         }
126                         this.selector = selector = null;
127                         // The method will return to the caller at this point.
128                     }
129                 }
130             } else {
131                 // Use the existing selector if this worker has been started.
132                 selector = this.selector;
133             }
134 
135             assert selector != null && selector.isOpen();
136 
137             started = true;
138             boolean offered = registerTaskQueue.offer(registerTask);
139             assert offered;
140         }
141 
142         if (wakenUp.compareAndSet(false, true)) {
143             selector.wakeup();
144         }
145     }

147     public void run() {
148         thread = Thread.currentThread();
149 
150         boolean shutdown = false;
151         Selector selector = this.selector;
152         for (;;) {
153             wakenUp.set(false);
154 
155             if (CONSTRAINT_LEVEL != 0) {
156                 selectorGuard.writeLock().lock();
157                     // This empty synchronization block prevents the selector
158                     // from acquiring its lock.
159                 selectorGuard.writeLock().unlock();
160             }
161 
162             try {
163                 SelectorUtil.select(selector);
164 
165                 // 'wakenUp.compareAndSet(false, true)' is always evaluated
166                 // before calling 'selector.wakeup()' to reduce the wake-up
167                 // overhead. (Selector.wakeup() is an expensive operation.)
168                 //
169                 // However, there is a race condition in this approach.
170                 // The race condition is triggered when 'wakenUp' is set to
171                 // true too early.
172                 //
173                 // 'wakenUp' is set to true too early if:
174                 // 1) Selector is waken up between 'wakenUp.set(false)' and
175                 //    'selector.select(...)'. (BAD)
176                 // 2) Selector is waken up between 'selector.select(...)' and
177                 //    'if (wakenUp.get()) { ... }'. (OK)
178                 //
179                 // In the first case, 'wakenUp' is set to true and the
180                 // following 'selector.select(...)' will wake up immediately.
181                 // Until 'wakenUp' is set to false again in the next round,
182                 // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
183                 // any attempt to wake up the Selector will fail, too, causing
184                 // the following 'selector.select(...)' call to block
185                 // unnecessarily.
186                 //
187                 // To fix this problem, we wake up the selector again if wakenUp
188                 // is true immediately after selector.select(...).
189                 // It is inefficient in that it wakes up the selector for both
190                 // the first case (BAD - wake-up required) and the second case
191                 // (OK - no wake-up required).
192 
193                 if (wakenUp.get()) {
194                     selector.wakeup();
195                 }
196 
197                 cancelledKeys = 0;
198                 processRegisterTaskQueue();
199                 processWriteTaskQueue();
200                 processSelectedKeys(selector.selectedKeys());
201 
202                 // Exit the loop when there's nothing to handle.
203                 // The shutdown flag is used to delay the shutdown of this
204                 // loop to avoid excessive Selector creation when
205                 // connections are registered in a one-by-one manner instead of
206                 // concurrent manner.
207                 if (selector.keys().isEmpty()) {
208                     if (shutdown ||
209                         executor instanceof ExecutorService && ((ExecutorService) executor).isShutdown()) {
210 
211                         synchronized (startStopLock) {
212                             if (registerTaskQueue.isEmpty() && selector.keys().isEmpty()) {
213                                 started = false;
214                                 try {
215                                     selector.close();
216                                 } catch (IOException e) {
217                                     logger.warn(
218                                             "Failed to close a selector.", e);
219                                 } finally {
220                                     this.selector = null;
221                                 }
222                                 break;
223                             } else {
224                                 shutdown = false;
225                             }
226                         }
227                     } else {
228                         // Give one more second.
229                         shutdown = true;
230                     }
231                 } else {
232                     shutdown = false;
233                 }
234             } catch (Throwable t) {
235                 logger.warn(
236                         "Unexpected exception in the selector loop.", t);
237 
238                 // Prevent possible consecutive immediate failures that lead to
239                 // excessive CPU consumption.
240                 try {
241                     Thread.sleep(1000);
242                 } catch (InterruptedException e) {
243                     // Ignore.
244                 }
245             }
246         }
247     }

249     private void processRegisterTaskQueue() throws IOException {
250         for (;;) {
251             final Runnable task = registerTaskQueue.poll();
252             if (task == null) {
253                 break;
254             }
255 
256             task.run();
257             cleanUpCancelledKeys();
258         }
259     }
260 
261     private void processWriteTaskQueue() throws IOException {
262         for (;;) {
263             final Runnable task = writeTaskQueue.poll();
264             if (task == null) {
265                 break;
266             }
267 
268             task.run();
269             cleanUpCancelledKeys();
270         }
271     }

273     private void processSelectedKeys(Set<SelectionKey> selectedKeys) throws IOException {
274         for (Iterator<SelectionKey> i = selectedKeys.iterator(); i.hasNext();) {
275             SelectionKey k = i.next();
276             i.remove();
277             try {
278                 int readyOps = k.readyOps();
279                 if ((readyOps & SelectionKey.OP_READ) != 0 || readyOps == 0) {
280                     if (!read(k)) {
281                         // Connection already closed - no need to handle write.
282                         continue;
283                     }
284                 }
285                 if ((readyOps & SelectionKey.OP_WRITE) != 0) {
286                     writeFromSelectorLoop(k);
287                 }
288             } catch (CancelledKeyException e) {
289                 close(k);
290             }
291 
292             if (cleanUpCancelledKeys()) {
293                 break; // break the loop to avoid ConcurrentModificationException
294             }
295         }
296     }
297 
298     private boolean cleanUpCancelledKeys() throws IOException {
299         if (cancelledKeys >= CLEANUP_INTERVAL) {
300             cancelledKeys = 0;
301             selector.selectNow();
302             return true;
303         }
304         return false;
305     }

307     private boolean read(SelectionKey k) {
308         final SocketChannel ch = (SocketChannel) k.channel();
309         final NioSocketChannel channel = (NioSocketChannel) k.attachment();
310 
311         final ReceiveBufferSizePredictor predictor =
312             channel.getConfig().getReceiveBufferSizePredictor();
313         final int predictedRecvBufSize = predictor.nextReceiveBufferSize();
314 
315         int ret = 0;
316         int readBytes = 0;
317         boolean failure = true;
318 
319         ByteBuffer bb = recvBufferPool.acquire(predictedRecvBufSize);
320         try {
321             while ((ret = ch.read(bb)) > 0) {
322                 readBytes += ret;
323                 if (!bb.hasRemaining()) {
324                     break;
325                 }
326             }
327             failure = false;
328         } catch (ClosedChannelException e) {
329             // Can happen, and does not need a user attention.
330         } catch (Throwable t) {
331             fireExceptionCaught(channel, t);
332         }
333 
334         if (readBytes > 0) {
335             bb.flip();
336 
337             final ChannelBufferFactory bufferFactory =
338                 channel.getConfig().getBufferFactory();
339             final ChannelBuffer buffer = bufferFactory.getBuffer(readBytes);
340             buffer.setBytes(0, bb);
341             buffer.writerIndex(readBytes);
342 
343             recvBufferPool.release(bb);
344 
345             // Update the predictor.
346             predictor.previousReceiveBufferSize(readBytes);
347 
348             // Fire the event.
349             fireMessageReceived(channel, buffer);
350         } else {
351             recvBufferPool.release(bb);
352         }
353 
354         if (ret < 0 || failure) {
355             close(channel, succeededFuture(channel));
356             return false;
357         }
358 
359         return true;
360     }

362     private void close(SelectionKey k) {
363         NioSocketChannel ch = (NioSocketChannel) k.attachment();
364         close(ch, succeededFuture(ch));
365     }

367     void writeFromUserCode(final NioSocketChannel channel) {
368         if (!channel.isConnected()) {
369             cleanUpWriteBuffer(channel);
370             return;
371         }
372 
373         if (scheduleWriteIfNecessary(channel)) {
374             return;
375         }
376 
377         // From here, we are sure Thread.currentThread() == workerThread.
378 
379         if (channel.writeSuspended) {
380             return;
381         }
382 
383         if (channel.inWriteNowLoop) {
384             return;
385         }
386 
387         write0(channel);
388     }

390     void writeFromTaskLoop(final NioSocketChannel ch) {
391         if (!ch.writeSuspended) {
392             write0(ch);
393         }
394     }

396     void writeFromSelectorLoop(final SelectionKey k) {
397         NioSocketChannel ch = (NioSocketChannel) k.attachment();
398         ch.writeSuspended = false;
399         write0(ch);
400     }

402     private boolean scheduleWriteIfNecessary(final NioSocketChannel channel) {
403         final Thread currentThread = Thread.currentThread();
404         final Thread workerThread = thread;
405         if (currentThread != workerThread) {
406             if (channel.writeTaskInTaskQueue.compareAndSet(false, true)) {
407                 boolean offered = writeTaskQueue.offer(channel.writeTask);
408                 assert offered;
409             }
410 
411             if (!(channel instanceof NioAcceptedSocketChannel) ||
412                 ((NioAcceptedSocketChannel) channel).bossThread != currentThread) {
413                 final Selector workerSelector = selector;
414                 if (workerSelector != null) {
415                     if (wakenUp.compareAndSet(false, true)) {
416                         workerSelector.wakeup();
417                     }
418                 }
419             } else {
420                 // A write request can be made from an acceptor thread (boss)
421                 // when a user attempted to write something in:
422                 //
423                 //   * channelOpen()
424                 //   * channelBound()
425                 //   * channelConnected().
426                 //
427                 // In this case, there's no need to wake up the selector because
428                 // the channel is not even registered yet at this moment.
429             }
430 
431             return true;
432         }
433 
434         return false;
435     }

437     private void write0(NioSocketChannel channel) {
438         boolean open = true;
439         boolean addOpWrite = false;
440         boolean removeOpWrite = false;
441 
442         long writtenBytes = 0;
443 
444         final SocketSendBufferPool sendBufferPool = this.sendBufferPool;
445         final SocketChannel ch = channel.socket;
446         final Queue<MessageEvent> writeBuffer = channel.writeBuffer;
447         final int writeSpinCount = channel.getConfig().getWriteSpinCount();
448         synchronized (channel.writeLock) {
449             channel.inWriteNowLoop = true;
450             for (;;) {
451                 MessageEvent evt = channel.currentWriteEvent;
452                 SendBuffer buf;
453                 if (evt == null) {
454                     if ((channel.currentWriteEvent = evt = writeBuffer.poll()) == null) {
455                         removeOpWrite = true;
456                         channel.writeSuspended = false;
457                         break;
458                     }
459 
460                     channel.currentWriteBuffer = buf = sendBufferPool.acquire(evt.getMessage());
461                 } else {
462                     buf = channel.currentWriteBuffer;
463                 }
464 
465                 ChannelFuture future = evt.getFuture();
466                 try {
467                     long localWrittenBytes = 0;
468                     for (int i = writeSpinCount; i > 0; i --) {
469                         localWrittenBytes = buf.transferTo(ch);
470                         if (localWrittenBytes != 0) {
471                             writtenBytes += localWrittenBytes;
472                             break;
473                         }
474                         if (buf.finished()) {
475                             break;
476                         }
477                     }
478 
479                     if (buf.finished()) {
480                         // Successful write - proceed to the next message.
481                         buf.release();
482                         channel.currentWriteEvent = null;
483                         channel.currentWriteBuffer = null;
484                         evt = null;
485                         buf = null;
486                         future.setSuccess();
487                     } else {
488                         // Not written fully - perhaps the kernel buffer is full.
489                         addOpWrite = true;
490                         channel.writeSuspended = true;
491 
492                         if (localWrittenBytes > 0) {
493                             // Notify progress listeners if necessary.
494                             future.setProgress(
495                                     localWrittenBytes,
496                                     buf.writtenBytes(), buf.totalBytes());
497                         }
498                         break;
499                     }
500                 } catch (AsynchronousCloseException e) {
501                     // Doesn't need a user attention - ignore.
502                 } catch (Throwable t) {
503                     buf.release();
504                     channel.currentWriteEvent = null;
505                     channel.currentWriteBuffer = null;
506                     buf = null;
507                     evt = null;
508                     future.setFailure(t);
509                     fireExceptionCaught(channel, t);
510                     if (t instanceof IOException) {
511                         open = false;
512                         close(channel, succeededFuture(channel));
513                     }
514                 }
515             }
516             channel.inWriteNowLoop = false;
517         }
518 
519         fireWriteComplete(channel, writtenBytes);
520 
521         if (open) {
522             if (addOpWrite) {
523                 setOpWrite(channel);
524             } else if (removeOpWrite) {
525                 clearOpWrite(channel);
526             }
527         }
528     }

530     private void setOpWrite(NioSocketChannel channel) {
531         Selector selector = this.selector;
532         SelectionKey key = channel.socket.keyFor(selector);
533         if (key == null) {
534             return;
535         }
536         if (!key.isValid()) {
537             close(key);
538             return;
539         }
540 
541         // interestOps can change at any time and at any thread.
542         // Acquire a lock to avoid possible race condition.
543         synchronized (channel.interestOpsLock) {
544             int interestOps = channel.getRawInterestOps();
545             if ((interestOps & SelectionKey.OP_WRITE) == 0) {
546                 interestOps |= SelectionKey.OP_WRITE;
547                 key.interestOps(interestOps);
548                 channel.setRawInterestOpsNow(interestOps);
549             }
550         }
551     }
552 
553     private void clearOpWrite(NioSocketChannel channel) {
554         Selector selector = this.selector;
555         SelectionKey key = channel.socket.keyFor(selector);
556         if (key == null) {
557             return;
558         }
559         if (!key.isValid()) {
560             close(key);
561             return;
562         }
563 
564         // interestOps can change at any time and at any thread.
565         // Acquire a lock to avoid possible race condition.
566         synchronized (channel.interestOpsLock) {
567             int interestOps = channel.getRawInterestOps();
568             if ((interestOps & SelectionKey.OP_WRITE) != 0) {
569                 interestOps &= ~SelectionKey.OP_WRITE;
570                 key.interestOps(interestOps);
571                 channel.setRawInterestOpsNow(interestOps);
572             }
573         }
574     }

576     void close(NioSocketChannel channel, ChannelFuture future) {
577         boolean connected = channel.isConnected();
578         boolean bound = channel.isBound();
579         try {
580             channel.socket.close();
581             cancelledKeys ++;
582 
583             if (channel.setClosed()) {
584                 future.setSuccess();
585                 if (connected) {
586                     fireChannelDisconnected(channel);
587                 }
588                 if (bound) {
589                     fireChannelUnbound(channel);
590                 }
591 
592                 cleanUpWriteBuffer(channel);
593                 fireChannelClosed(channel);
594             } else {
595                 future.setSuccess();
596             }
597         } catch (Throwable t) {
598             future.setFailure(t);
599             fireExceptionCaught(channel, t);
600         }
601     }

603     private void cleanUpWriteBuffer(NioSocketChannel channel) {
604         Exception cause = null;
605         boolean fireExceptionCaught = false;
606 
607         // Clean up the stale messages in the write buffer.
608         synchronized (channel.writeLock) {
609             MessageEvent evt = channel.currentWriteEvent;
610             if (evt != null) {
611                 // Create the exception only once to avoid the excessive overhead
612                 // caused by fillStackTrace.
613                 if (channel.isOpen()) {
614                     cause = new NotYetConnectedException();
615                 } else {
616                     cause = new ClosedChannelException();
617                 }
618 
619                 ChannelFuture future = evt.getFuture();
620                 channel.currentWriteBuffer.release();
621                 channel.currentWriteBuffer = null;
622                 channel.currentWriteEvent = null;
623                 evt = null;
624                 future.setFailure(cause);
625                 fireExceptionCaught = true;
626             }
627 
628             Queue<MessageEvent> writeBuffer = channel.writeBuffer;
629             if (!writeBuffer.isEmpty()) {
630                 // Create the exception only once to avoid the excessive overhead
631                 // caused by fillStackTrace.
632                 if (cause == null) {
633                     if (channel.isOpen()) {
634                         cause = new NotYetConnectedException();
635                     } else {
636                         cause = new ClosedChannelException();
637                     }
638                 }
639 
640                 for (;;) {
641                     evt = writeBuffer.poll();
642                     if (evt == null) {
643                         break;
644                     }
645                     evt.getFuture().setFailure(cause);
646                     fireExceptionCaught = true;
647                 }
648             }
649         }
650 
651         if (fireExceptionCaught) {
652             fireExceptionCaught(channel, cause);
653         }
654     }

656     void setInterestOps(
657             NioSocketChannel channel, ChannelFuture future, int interestOps) {
658         boolean changed = false;
659         try {
660             // interestOps can change at any time and at any thread.
661             // Acquire a lock to avoid possible race condition.
662             synchronized (channel.interestOpsLock) {
663                 Selector selector = this.selector;
664                 SelectionKey key = channel.socket.keyFor(selector);
665 
666                 if (key == null || selector == null) {
667                     // Not registered to the worker yet.
668                     // Set the rawInterestOps immediately; RegisterTask will pick it up.
669                     channel.setRawInterestOpsNow(interestOps);
670                     return;
671                 }
672 
673                 // Override OP_WRITE flag - a user cannot change this flag.
674                 interestOps &= ~Channel.OP_WRITE;
675                 interestOps |= channel.getRawInterestOps() & Channel.OP_WRITE;
676 
677                 switch (CONSTRAINT_LEVEL) {
678                 case 0:
679                     if (channel.getRawInterestOps() != interestOps) {
680                         key.interestOps(interestOps);
681                         if (Thread.currentThread() != thread &&
682                             wakenUp.compareAndSet(false, true)) {
683                             selector.wakeup();
684                         }
685                         changed = true;
686                     }
687                     break;
688                 case 1:
689                 case 2:
690                     if (channel.getRawInterestOps() != interestOps) {
691                         if (Thread.currentThread() == thread) {
692                             key.interestOps(interestOps);
693                             changed = true;
694                         } else {
695                             selectorGuard.readLock().lock();
696                             try {
697                                 if (wakenUp.compareAndSet(false, true)) {
698                                     selector.wakeup();
699                                 }
700                                 key.interestOps(interestOps);
701                                 changed = true;
702                             } finally {
703                                 selectorGuard.readLock().unlock();
704                             }
705                         }
706                     }
707                     break;
708                 default:
709                     throw new Error();
710                 }
711 
712                 if (changed) {
713                     channel.setRawInterestOpsNow(interestOps);
714                 }
715             }
716 
717             future.setSuccess();
718             if (changed) {
719                 fireChannelInterestChanged(channel);
720             }
721         } catch (CancelledKeyException e) {
722             // setInterestOps() was called on a closed channel.
723             ClosedChannelException cce = new ClosedChannelException();
724             future.setFailure(cce);
725             fireExceptionCaught(channel, cce);
726         } catch (Throwable t) {
727             future.setFailure(t);
728             fireExceptionCaught(channel, t);
729         }
730     }

732     private final class RegisterTask implements Runnable {
733         private final NioSocketChannel channel;
734         private final ChannelFuture future;
735         private final boolean server;
736 
737         RegisterTask(
738                 NioSocketChannel channel, ChannelFuture future, boolean server) {
739 
740             this.channel = channel;
741             this.future = future;
742             this.server = server;
743         }
744 
745         public void run() {
746             SocketAddress localAddress = channel.getLocalAddress();
747             SocketAddress remoteAddress = channel.getRemoteAddress();
748             if (localAddress == null || remoteAddress == null) {
749                 if (future != null) {
750                     future.setFailure(new ClosedChannelException());
751                 }
752                 close(channel, succeededFuture(channel));
753                 return;
754             }
755 
756             try {
757                 if (server) {
758                     channel.socket.configureBlocking(false);
759                 }
760 
761                 synchronized (channel.interestOpsLock) {
762                     channel.socket.register(
763                             selector, channel.getRawInterestOps(), channel);
764                 }
765                 if (future != null) {
766                     channel.setConnected();
767                     future.setSuccess();
768                 }
769             } catch (IOException e) {
770                 if (future != null) {
771                     future.setFailure(e);
772                 }
773                 close(channel, succeededFuture(channel));
774                 if (!(e instanceof ClosedChannelException)) {
775                     throw new ChannelException(
776                             "Failed to register a socket to the selector.", e);
777                 }
778             }
