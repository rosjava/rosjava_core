54      static int crc32(int crc32, byte[] buf, int index, int len) {
55          final int endIndex = index + len;
56          crc32 ^= 0xffffffff;
57          for (int i = index; i < endIndex; i ++) {
58              crc32 = crc32 >>> 8 ^ TABLE[(crc32 ^ buf[i]) & 0xff];
59          }
60          crc32 ^= 0xffffffff;
61          return crc32;
62      }
